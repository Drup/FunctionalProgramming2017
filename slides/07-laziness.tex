\begin{frame}
  \frametitle{Laziness}
  \begin{itemize}
  \item Haskell does not evaluate function arguments unless they are
    demanded to calculate the function's result
  \item Data structures are only evaluated as much as they are needed
  \end{itemize}
\end{frame}
